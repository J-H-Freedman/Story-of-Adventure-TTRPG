%%% newplayr 1: Text
%%% Sets up a figure box for callouts for new players
\def\newplayr#1{%
	\begin{figure}[H]{%
			\makebox[3em]{\dotfill}\ {\large New Players!}\ \dotfill\vspace{-1em}\linebreak
			\begin{flushleft}
				#1\vspace{-1em}\linebreak
			\end{flushleft}
			\strut\dotfill
			}
	\end{figure}
	}

\def\notetoGM#1{%
	\begin{figure}[H]{%
			\makebox[3em]{\hrulefill}\ {\large A Note to the GM}\ \hrulefill\vspace{-2em}\linebreak
			\begin{flushleft}
				#1\vspace{-2em}\linebreak
			\end{flushleft}
			\strut\hrulefill
			}
	\end{figure}
	}
